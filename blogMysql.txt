CREATE DATABASE blog;

CREATE TABLE t_blog(
id BIGINT AUTO_INCREMENT PRIMARY KEY,
title VARCHAR(255),
content LONGTEXT,
firstPic VARCHAR(255),
flag VARCHAR(255),
views INT,
appreciation BIT(1) NOT NULL,
shareStatement BIT(1) NOT NULL,
commentAble BIT(1) NOT NULL,
publish BIT(1) NOT NULL,
recommend BIT(1) NOT NULL,
createTime DATETIME,
updateTime DATETIME,
description VARCHAR(255),
typeId BIGINT,
userId BIGINT
)ENGINE=INNODB AUTO_INCREMENT=1 DEFAULT CHARSET=utf8;

CREATE TABLE t_comment(
id BIGINT AUTO_INCREMENT PRIMARY KEY,
nickname VARCHAR(255),
email VARCHAR(255),
content VARCHAR(255),
headPic VARCHAR(255),
createTime DATETIME,
blogId BIGINT,
parentCommentId BIGINT,
adminComment BIT(1) NOT NULL
)ENGINE=INNODB AUTO_INCREMENT=1 DEFAULT CHARSET=utf8;

CREATE TABLE t_tag(
gId BIGINT AUTO_INCREMENT PRIMARY KEY,
name VARCHAR(255),
blogId BIGINT
)ENGINE=INNODB AUTO_INCREMENT=1 DEFAULT CHARSET=utf8;

CREATE TABLE t_type(
tId BIGINT AUTO_INCREMENT PRIMARY KEY,
name VARCHAR(255),
blogId BIGINT
)ENGINE=INNODB AUTO_INCREMENT=1 DEFAULT CHARSET=utf8;

CREATE TABLE user(
id BIGINT AUTO_INCREMENT PRIMARY KEY,
username VARCHAR(255),
password VARCHAR(255),
email VARCHAR(255),
authority VARCHAR(255),
headPic VARCHAR(255),
createTime DATETIME,
updateTime DATETIME
)ENGINE=INNODB AUTO_INCREMENT=1 DEFAULT CHARSET=utf8;

CREATE TABLE t_blog_tag(
blogId BIGINT,
tagId BIGINT
)ENGINE=INNODB DEFAULT CHARSET=utf8;
